library(MASS) library(rpart) library(rpart.plot) summary(birthwt) mydf <- birthwt mydf_new <- mydf[,-which(names(mydf) == "bwt")] mytree <- rpart(low~.,data= mydf_new,method="class", control = rpart.control(cp=0.05)) rpart.plot::rpart.plot(mytree, type = 1, extra=1, box.palette =c("pink", "green"), branch.lty=3, shadow.col = "gray") par(mar=c(1,1,1,1)) c)The Tree is very interesting, it shows that if the age of the mother is more than 23 then the baby will have weight less than 2,5 kg.
